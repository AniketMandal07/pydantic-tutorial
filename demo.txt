
def insert_patient_data(name: str, age: int):
    
    if type(name) == str and type(age) == int:
        if age < 0:
            raise ValueError("Age cant be negative")
        else:
            print(name)
            print(age)
            print('insert into database')
    else:
        raise TypeError('Incorrect data type')
